package entities;

import ultilities.Vector2;

public enum Direction {
    UP(0.0f, -1.0f),
    RIGHT(1.0f, 0.0f),
    DOWN(0.0f, 1.0f),
    LEFT(-1.0f, 0.0f),
    NONE(0.0f, 0.0f);
    private final Vector2 direction;

    Direction(float x, float y) {
        this.direction = new Vector2(x, y);
    }

    public Vector2 getDirection() {
        return direction;
    }
}
