     public int buttonPressed(){
         for(int i = 0; i<x.length; i++){
             if(Main.processing.mouseX > x[i] && Main.processing.mouseX < x[i]+w[i] &&
                     Main.processing.mouseY > y[i] && Main.processing.mouseY < y[i]+h[i]){
                 return i;
             }
         }
         return -1;
     }
